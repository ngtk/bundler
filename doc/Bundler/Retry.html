def initialize(name, exceptions = nil, retries = self.class.default_retries)
  @name = name
  @retries = retries
  @exceptions = Array(exceptions) || []
  @total_runs = @retries + 1 # will run once, then upto attempts.times
end

def attempt(&block)
  @current_run = 0
  @failed      = false
  @error       = nil
  run(&block) while keep_trying?
  @result
end
